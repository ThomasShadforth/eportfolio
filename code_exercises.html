              from abc import ABC, abstractmethod
              import math

              class Shape(ABC):
                  @abstractmethod
                  def area(self) -> float:

                    pass

              class Circle(Shape):
                  def __init__(self, radius) -> None:
                      self.radius = radius

                  def area(self) -> float:
  
                      return math.pi * (self.radius ** 2)

              class Rectangle(Shape):

                  def __init__(self, length, width) -> None:
                      self.length = length

                      self.width = width

                  def area(self) -> float:

                      return self.length * self.width

            from abc import ABC, abstractmethod

            #Define abstract class for payment method
            class PaymentMethod(ABC):
                #define abstract method "pay"
                @abstractmethod
                def pay(self) -> None:
                    pass

            class ShoppingItem():
                def __init__(self, name: str, price: float) -> None:
                    self.name = name
                    self.price = price

            # Define concrete implementations of the abstract paymentmethod class
            class CreditCard(PaymentMethod):

                def pay(self) -> None:
                    print("Processing Credit Card Payment...")

            class PayPal(PaymentMethod):

                def pay(self) -> None:
                    print("Processing PayPal Payment...")

            class CryptoMethod(PaymentMethod):

                def pay(self) -> None:
                    print("Processing Crypto-Currency Payment...")

            class PaymentProcessor:

                def pay(self, payment_type: PaymentMethod) -> None:
                    payment_type.pay()

            class Order:
                #Separated Order into Order and PaymentProcessor
                def __init__(self) -> None:
                    self.items = []

                def add_item(self, item: ShoppingItem) -> None:
                    self.items.append(item)

                def calculate_total(self) -> float:

                    return sum(item.price for item in self.items)

            # Create instance of creditcard, and instance of paymentprocessor
            my_payment_method = CreditCard()
            pay_processor = PaymentProcessor()

            # Define a set of shoppingitem instances
            apple = ShoppingItem("Apple", 0.55)
            orange = ShoppingItem("orange", 0.73)
            cereal = ShoppingItem("cereal", 1.55)

            # Create a new order
            new_order = Order()

            # Add Shopping Items
            new_order.add_item(apple)
            new_order.add_item(orange)
            new_order.add_item(cereal)

            # Calculate Total Cost of Shopping Order
            print("Total cost of shopping is: £" + str(new_order.calculate_total()))

            # Pay using the established payment method
            pay_processor.pay(my_payment_method)

Implementing the Factory Pattern

          
            from abc import ABC, abstractmethod

            class Car(ABC):

                @abstractmethod
                def drive(self) -> None:
                    pass

            class Sedan(Car):

                def drive(self) -> None:
                    print("Driving a Sedan")

            class SUV(Car):
                def drive(self) -> None:
                    print("Driving an SUV")

            class Hatchback(Car):
                def drive(self) -> None:
                    print("Driving a Hatchback")

            class CarFactory(ABC):
                @abstractmethod
                def create_car(self) -> Car:
                    pass

            class SedanFactory(CarFactory):
                def create_car(self) -> Car:
                    return Sedan()

            class SUVFactory(CarFactory):
                def create_car(self) -> Car:
                    return SUV()

            class HatchbackFactory(CarFactory):
                def create_car(self) -> Car:
                    return Hatchback()

            def client_code(car_factory : CarFactory) -> None:
                car = car_factory.create_car()
                car.drive()

            if __name__ == '__main__':
                print("Creating car using Sedan Factory")
                client_code(SedanFactory())

                print("Creating car using SUV Factory")
                client_code(SUVFactory())

                print("Creating car using Hatchback Factory")
                client_code(HatchbackFactory())

              import threading
              from concurrent.futures import ThreadPoolExecutor

              class BankAccount:
                  def __init__(self, account_number, balance, is_safe=True) -> None:
                      self.account_number = account_number
                      self.__balance = balance
                      self.lock = threading.Lock()
                      self.__is_safe = is_safe

                  def deposit(self, amount: float) -> None:
                      if self.__is_safe:
                          with self.lock:
                              self.__balance = self.__balance + amount
                      else:
                          self.__balance = self.__balance + amount
                  
                  def withdraw(self, withdraw_amount: float) -> None:
                      if self.__is_safe:
                          with self.lock:
                              balance_after_withdraw = self.__balance - withdraw_amount

                          if 0.0 > balance_after_withdraw:
                              print("INSUFFICIENT FUNDS AVAILABLE, CANNOT WITHDRAW: £" + str(withdraw_amount))
                          else:
                              self.__balance = balance_after_withdraw
                      else:
                          balance_after_withdraw = self.__balance - withdraw_amount

                          if 0.0 > balance_after_withdraw:
                              print("INSUFFICIENT FUNDS AVAILABLE, CANNOT WITHDRAW: £" + str(withdraw_amount))
                          else:
                              self.__balance = balance_after_withdraw
                  def print_balance(self) -> None:
                      if self.__is_safe:
                          with self.lock:
                              print("Current balance: £" + str(self.__balance))
                      else:
                          print("Current balance: £" + str(self.__balance))
              
                  def get_balance(self) -> float:
                      if self.__is_safe:
                          with self.lock:
                              return self.__balance
                      else:
                          return self.__balance

              class TransactionSimulator:

                  # Upon initialisation, assign passed bankaccount objects to member variables, and create a lock
                  def __init__(self, bank_account_1: BankAccount, bank_account_2: BankAccount, is_safe = True) -> None:
                      self.bank_account_1 = bank_account_1
                      self.bank_account_2 = bank_account_2
                      self.lock = threading.Lock()
                      # __is_safe is for testing purposes, will remain true by default
                      # if a transactionsimulator object has this marked as false, thread safety will not be applied
                      self.__is_safe = is_safe
              
                  # Perform multiple deposits and withdrawals in sequence. While defined as floats, TPEs will allow for iterables containing matching data types to be passed into this method
                  def SimulateDepositWithdraw(self, first_deposit_amount, first_withdrawal_amount, second_deposit_amount, second_withdrawal_amount) -> None:
                      if self.__is_safe:
                          with self.lock:
                              self.bank_account_1.deposit(first_deposit_amount)
                              self.bank_account_1.withdraw(first_withdrawal_amount)
                              self.bank_account_1.deposit(second_deposit_amount)
                              self.bank_account_1.withdraw(second_withdrawal_amount)
                      else:
                          self.bank_account_1.deposit(first_deposit_amount)
                          self.bank_account_1.withdraw(first_withdrawal_amount)
                          self.bank_account_1.deposit(second_deposit_amount)
                          self.bank_account_1.withdraw(second_withdrawal_amount)
              
                  # Transfer money between two accounts
                  def SimulateBankTransfer(self, sender_account, recipient_account, transfer_amount) -> None:
                      if self.__is_safe:
                          with self.lock:
                              sender_account.withdraw(transfer_amount)
                              recipient_account.deposit(transfer_amount)
                              print("Successfully transferred £" + str(transfer_amount) + " from account " + str(sender_account.account_number) + " to account " + str(recipient_account.account_number))
                      else:
                          sender_account.withdraw(transfer_amount)
                          recipient_account.deposit(transfer_amount)
                          print("Successfully transferred £" + str(transfer_amount) + " from account " + str(sender_account.account_number) + " to account " + str(recipient_account.account_number))
              
                  # Simulate a single deposit
                  def SimulateDeposit(self, deposit_amount) -> None:
                      if self.__is_safe:
                          with self.lock:
                              self.bank_account_1.deposit(deposit_amount)
                      else:
                          self.bank_account_1.deposit(deposit_amount)
              
                  # Simulate a single withdrawal
                  def SimulateWithdrawal(self, withdrawal_amount) -> None:
                      if self.__is_safe:
                          with self.lock:
                              self.bank_account_2.withdraw(withdrawal_amount)
                      else:
                          self.bank_account_2.withdraw(withdrawal_amount)
              
              # Will only run this code if this module is not being accessed outside of this file. E.g. will not run in the unittest file
              if __name__ == '__main__':
                  # Initialise the transaction simulator, passing in initialisation arguments for the two BankAccount objects (account_number: 12231253, balance: 300), (account_number: 32245799, balance: 1000)
                  transactSim = TransactionSimulator(BankAccount(12231253, 300), BankAccount(32245799, 1000))
              
                  # Initialise arrays for first deposit, second deposit, first withdrawal and second withdrawal amounts for use with thread pool executor
                  first_deposit_amounts = [30.51, 2.56, 90.32]
                  second_deposit_amounts = [300.21, 600.11, 425.77]
                  first_withdrawal_amounts = [99.12, 467.18, 1000.33]
                  second_withdrawal_amounts = [322.01, 3.12, 9.43]
              
                  # Initialise empty array called result, which will be used to store the output from the ThreadPoolExecutor call of the depositwithdraw method (And also the bank transfer method
                  result=[]
              
                  print("Simulating concurrent transactions on single account...")
              
                  # use thread pool executor to run concurrent calls of the transaction simulator's SimulateDepositWithdraw method
                  with ThreadPoolExecutor(max_workers=3) as executor:
                      # Pass in the defined arrays as parameters when mapping the function to threads, will determine the value used based on the thread.
                      result = executor.map(transactSim.SimulateDepositWithdraw, first_deposit_amounts, first_withdrawal_amounts, second_deposit_amounts, second_withdrawal_amounts)
                  for r in result:
                      # Iterate through result array and print output of ThreadPoolExecutor
                      if r != None:
                          print(r)
              
                  print("Simulating transfer between two bank accounts...")
              
                  # use thread pool executor to run concurrent calls of the transaction simulator's SimulateBankTransfer method
                  # Pass in the bank accounts as both a sender and recipient dependent on the thread in order to check for deadlocks
                  with ThreadPoolExecutor(max_workers=2) as executor:
                      result = executor.map(transactSim.SimulateBankTransfer, [transactSim.bank_account_1, transactSim.bank_account_2], [transactSim.bank_account_2, transactSim.bank_account_1], [300.20, 400])
                  for r in result:
                      if r != None:
                          print(r)
