              from abc import ABC, abstractmethod
              import math

              class Shape(ABC):
                  @abstractmethod
                  def area(self) -> float:

                    pass

              class Circle(Shape):
                  def __init__(self, radius) -> None:
                      self.radius = radius

                  def area(self) -> float:
  
                      return math.pi * (self.radius ** 2)

              class Rectangle(Shape):

                  def __init__(self, length, width) -> None:
                      self.length = length

                      self.width = width

                  def area(self) -> float:

                      return self.length * self.width

              circ1 = Circle(4.5)

              print(circ1.area())

              rect1 = rectangle(4, 6)

              print(rect1.area())

              class BankAccount:
                  def __init__(self, balance) -> None:
                      self.__balance = balance

                  def deposit(self, amount: float) -> None:
                      self.__balance += amount
                      print("£" + str(amount) + " has been deposited in account!")

                  def withdraw(self, amount: float) -> None:
                      fundsCheck = self.__balance - amount

                      if fundsCheck >= 0:
                          self.__balance -= amount
                          print("£" + str(amount) + " has been withdrawn from account!")
                      else:
                          print("INSUFFICIENT FUNDS")
                  
                  def balancecheck(self) -> float:

                      return self.__balance
              
              acc1 = BankAccount(200)

              acc1.deposit(35.50)
              acc1.withdraw(104.25)

              print("Account Balance is: £" + str(acc1.balancecheck())

Designing a Real World Application with SOLID Design Principles

          
            from abc import ABC, abstractmethod

            #Define abstract class for payment method
            class PaymentMethod(ABC):
                #define abstract method "pay"
                @abstractmethod
                def pay(self) -> None:
                    pass

            class ShoppingItem():
                def __init__(self, name: str, price: float) -> None:
                    self.name = name
                    self.price = price

            # Define concrete implementations of the abstract paymentmethod class
            class CreditCard(PaymentMethod):

                def pay(self) -> None:
                    print("Processing Credit Card Payment...")

            class PayPal(PaymentMethod):

                def pay(self) -> None:
                    print("Processing PayPal Payment...")

            class CryptoMethod(PaymentMethod):

                def pay(self) -> None:
                    print("Processing Crypto-Currency Payment...")

            class PaymentProcessor:

                def pay(self, payment_type: PaymentMethod) -> None:
                    payment_type.pay()

            class Order:
                #Separated Order into Order and PaymentProcessor
                def __init__(self) -> None:
                    self.items = []

                def add_item(self, item: ShoppingItem) -> None:
                    self.items.append(item)

                def calculate_total(self) -> float:

                    return sum(item.price for item in self.items)

            # Create instance of creditcard, and instance of paymentprocessor
            my_payment_method = CreditCard()
            pay_processor = PaymentProcessor()

            # Define a set of shoppingitem instances
            apple = ShoppingItem("Apple", 0.55)
            orange = ShoppingItem("orange", 0.73)
            cereal = ShoppingItem("cereal", 1.55)

            # Create a new order
            new_order = Order()

            # Add Shopping Items
            new_order.add_item(apple)
            new_order.add_item(orange)
            new_order.add_item(cereal)

            # Calculate Total Cost of Shopping Order
            print("Total cost of shopping is: £" + str(new_order.calculate_total()))

            # Pay using the established payment method
            pay_processor.pay(my_payment_method)

Implementing the Factory Pattern

          
            from abc import ABC, abstractmethod

            class Car(ABC):

                @abstractmethod
                def drive(self) -> None:
                    pass

            class Sedan(Car):

                def drive(self) -> None:
                    print("Driving a Sedan")

            class SUV(Car):
                def drive(self) -> None:
                    print("Driving an SUV")

            class Hatchback(Car):
                def drive(self) -> None:
                    print("Driving a Hatchback")

            class CarFactory(ABC):
                @abstractmethod
                def create_car(self) -> Car:
                    pass

            class SedanFactory(CarFactory):
                def create_car(self) -> Car:
                    return Sedan()

            class SUVFactory(CarFactory):
                def create_car(self) -> Car:
                    return SUV()

            class HatchbackFactory(CarFactory):
                def create_car(self) -> Car:
                    return Hatchback()

            def client_code(car_factory : CarFactory) -> None:
                car = car_factory.create_car()
                car.drive()

            if __name__ == '__main__':
                print("Creating car using Sedan Factory")
                client_code(SedanFactory())

                print("Creating car using SUV Factory")
                client_code(SUVFactory())

                print("Creating car using Hatchback Factory")
                client_code(HatchbackFactory())
